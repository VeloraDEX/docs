## Query parameters

<ParamField query="srcToken" type="string" required default="0xEeeeeEeeeEeEeeEeEeEeeEEEeeeeEeeeeeeeEEeE">
  Source token address.
</ParamField>

<ParamField query="srcDecimals" type="integer" required default={18}>
  Source token decimals.
</ParamField>

<ParamField query="destToken" type="string" required default="0xa0b86991c6218b36c1d19d4a2e9eb0ce3606eb48">
  Destination token address. For cross-chain quotes, the destination token on the destination chain.
</ParamField>

<ParamField query="destDecimals" type="integer" required default={6}>
  Destination token decimals.
</ParamField>

<ParamField query="amount" type="string" required default="1000000000000000000">
  `srcToken` amount (SELL) or `destToken` amount (BUY), in WEI/raw units.
</ParamField>

<ParamField query="side" type="string" default="SELL">
  `SELL` or `BUY`.
</ParamField>

<ParamField query="chainId" type="integer" required default={1}>
  Source chain ID. Supported values: `1` (Mainnet), `10` (Optimism), `56` (BSC), `137` (Polygon), `250` (Fantom), `1101` (zkEVM), `8453` (Base), `42161` (Arbitrum), `43114` (Avalanche), `100` (Gnosis).
</ParamField>

<ParamField query="mode" type="string" default="ALL">
  Controls pricing source and fallback behavior. `ALL` returns both Delta and Market blocks; `DELTA` returns Delta only; `MARKET` returns Market only.
</ParamField>

<ParamField query="userAddress" type="string" default="0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045">
  End-user wallet address.
</ParamField>

<ParamField query="partner" type="string" default="my-app-name">
  Partner string. Defaults to `anon` which charges 1bps fee for all swaps.
</ParamField>

<ParamField query="destChainId" type="integer">
  Destination chain ID for cross-chain quotes. Omit for same-chain quotes.
</ParamField>
