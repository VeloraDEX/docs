```go Go
package main

import (
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"net/url"
)

func main() {
	q := url.Values{}
	q.Set("srcToken",    "0xEeeeeEeeeEeEeeEeEeEeeEEEeeeeEeeeeeeeEEeE")
	q.Set("destToken",   "0xa0b86991c6218b36c1d19d4a2e9eb0ce3606eb48")
	q.Set("amount",      "1000000000000000000")
	q.Set("srcDecimals", "18")
	q.Set("destDecimals","6")
	q.Set("side",        "SELL")
	q.Set("chainId",     "1")
	q.Set("mode",        "ALL")
	q.Set("userAddress", "0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045")
	q.Set("partner",     "my-app-name")

	resp, _ := http.Get("https://api.velora.xyz/quote?" + q.Encode())
	defer resp.Body.Close()
	body, _ := io.ReadAll(resp.Body)

	var out map[string]any
	json.Unmarshal(body, &out)
	fmt.Println(out)
}
```
