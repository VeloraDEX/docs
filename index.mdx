```python Python
import requests

r = requests.get("https://api.paraswap.io/quote", params={

    "destToken": "0xdAC17F958D2ee523a2206206994597C13D831ec7",
    "amount":    "1000000000",
    "side":      "SELL",
    "network":   1,
    "mode":      "ALL",
}).json()
```
